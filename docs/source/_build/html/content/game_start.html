Старт игры¶

В центрифугу прилетит ответ от сервера о начале игры:

Ответ от сервера:

{
    "move": 1, // номер текущего хода
    "data": {
        "bank": {
            "total": 15, // общая ставка за все ходы
            "current": 15 // ставка за текущий этап игры
        },
        "players": [
            {
                "user": {
                    "id": 345345,
                    "chips": 50000, // всего фишек у данного пользователя в игре
                    "place": 0 // место за столом
                },
                "game": {
                    "bet": 0, // ставка игрока за данный этап игры
                    "isDealer": true // (* указывается только если) пользователь является дилером в данной игре
                }
            },
            {
                "user": {
                    "id": 456455,
                    "chips": 38000,
                    "place": 1
                },
                "game": {
                    "bet": 5,
                    "isSmallBlind": true // (* указывается только если) пользователь сделал small blind
                }
            },
            {
                "user": {
                    "id": 145344,
                    "chips": 43000,
                    "place": 2
                },
                "game": {
                    "bet": 10,
                    "isBigBlind": true // (* указывается только если) пользователь сделал big blind
                }
            }
        ],
        "game": {
            "move": 345345 // id пользователя, чей ход следующий
        }
    }
}
